LWC Forms & Validation

Building forms in LWC requires handling input, validation, and error display. Learn best practices for creating robust, user-friendly forms.

1. Basic Form Structure


<template>
    <div class="container">
        <h2>Account Form</h2>

        <form onsubmit={handleSubmit}>
            <div class="form-group">

                <button type="submit" class="btn-primary">Save</button>

                {successMessage}
            </div>

                {generalError}
            </div>

    </div>

2. Form Data Binding


import { LightningElement, track } from 'lwc';

export default class AccountForm extends LightningElement {
    @track formData = {
        name: '',
        phone: '',
        website: '',
        industry: ''
    };

    @track errors = {};
    @track successMessage = '';
    @track generalError = '';

    handleInputChange(event) {
        const field = event.target.dataset.field;
        const value = event.target.value;

        // Update formData
        this.formData = { ...this.formData, [field]: value };

        // Clear error for this field on change
        if (this.errors[field]) {
            this.errors = { ...this.errors, [field]: '' };
        }
    }

    handleSubmit(event) {
        event.preventDefault();

        // Validate form
        if (!this.validateForm()) {
            return;
        }

        // Submit form
        this.submitForm();
    }

    validateForm() {
        const errors = {};

        if (!this.formData.name || this.formData.name.trim() === '') {
            errors.name = 'Account name is required';
        }

        if (!this.formData.industry) {
            errors.industry = 'Industry is required';
        }

        if (this.formData.phone && !this.isValidPhone(this.formData.phone)) {
            errors.phone = 'Phone number is invalid';
        }

        this.errors = errors;
        return Object.keys(errors).length === 0;
    }

    isValidPhone(phone) {
        const phoneRegex = /^[\d\s\-()]+$/;
        return phoneRegex.test(phone) && phone.length >= 10;
    }

    async submitForm() {
        try {
            // Call Apex to save account
            const result = await createAccount({ 
                accountData: this.formData 
            });

            this.successMessage = 'Account saved successfully!';
            this.formData = { name: '', phone: '', website: '', industry: '' };

            // Clear message after 3 seconds
            setTimeout(() => {
                this.successMessage = '';
            }, 3000);
        } catch (error) {
            this.generalError = error.body.message;
        }
    }

    handleCancel() {
        this.formData = { name: '', phone: '', website: '', industry: '' };
        this.errors = {};
        this.successMessage = '';
        this.generalError = '';
    }
}

4. Real-time Validation


export default class AccountForm extends LightningElement {
    @track email = '';
    @track emailError = '';

    handleEmailChange(event) {
        this.email = event.target.value;
        this.validateEmail();
    }

    validateEmail() {
        const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

        if (!this.email) {
            this.emailError = '';
        } else if (!emailRegex.test(this.email)) {
            this.emailError = 'Please enter a valid email address';
        } else {
            this.emailError = '';
        }
    }
}

5. File Upload


<template>
    <form onsubmit={handleUpload}>
        <input 
            type="file" 
            accept=".csv,.xlsx"
            onchange={handleFileSelect}>

        <span if:true={fileName}>
            Selected file: {fileName}
        </span>

        <button type="submit" disabled={isSubmitting}>
            Upload
        </button>
    </form>
</template>

export default class FileUpload extends LightningElement {
    @track fileName = '';
    @track isSubmitting = false;
    selectedFile;

    handleFileSelect(event) {
        this.selectedFile = event.target.files[0];
        this.fileName = this.selectedFile.name;
    }

    async handleUpload(event) {
        event.preventDefault();

        if (!this.selectedFile) {
            return;
        }

        this.isSubmitting = true;

        try {
            const reader = new FileReader();
            reader.onload = async (e) => {
                const fileContent = e.target.result;
                await uploadFile({ 
                    fileName: this.fileName,
                    fileContent: fileContent
                });
            };
            reader.readAsArrayBuffer(this.selectedFile);
        } catch (error) {
            console.error('Upload failed:', error);
        } finally {
            this.isSubmitting = false;
        }
    }
}

6. Form Validation Patterns

Password Strength Validator


validatePasswordStrength(password) {
    const rules = {
        length: password.length >= 8,
        uppercase: /[A-Z]/.test(password),
        lowercase: /[a-z]/.test(password),
        numbers: /\d/.test(password),
        special: /[!@#$%^&*]/.test(password)
    };

    const strength = Object.values(rules).filter(Boolean).length;

    return {
        rules,
        strength,
        label: strength === 5 ? 'Strong' : strength >= 3 ? 'Medium' : 'Weak'
    };
}

Cross-Field Validation


validateDates() {
    if (this.startDate && this.endDate) {
        if (new Date(this.startDate) > new Date(this.endDate)) {
            this.errors.endDate = 'End date must be after start date';
            return false;
        }
    }
    return true;
}

validatePasswords() {
    if (this.password !== this.confirmPassword) {
        this.errors.confirmPassword = 'Passwords do not match';
        return false;
    }
    return true;
}

7. Error Display Best Practices


<template>
    <div class="form-group">
        <label>Email</label>
        <input 
            type="email"
            value={email}
            onchange={handleEmailChange}
            class={inputClass}>
        
        <div if:true={emailError} class="error-message">

                <!-- Error icon -->
            </svg>
            <span>{emailError}</span>
        </div>
    </div>
</template>

export default class EmailInput extends LightningElement {
    @track email = '';
    @track emailError = '';

    get inputClass() {
        return this.emailError ? 'input error' : 'input';
    }
}

Key Takeaways

  • Use @track for form data and errors
  • Validate on submit and in real-time
  • Clear errors when user starts correcting
  • Use Lightning components for consistent UI
  • Show success/error feedback clearly
  • Disable submit button while processing
  • Handle file uploads with FileReader API
